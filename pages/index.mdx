import { HomePage } from "@components/home";
import React from "react";
import { useSSG } from "nextra/ssg";
import { API_ENDPOINT } from "../src/config";

export async function getStaticProps() {
    const { approximate_member_count } = await fetch(
        "https://discord.com/api/v10/invites/yeecord?with_counts=true"
    ).then((r) => r.json());

    const info = await fetch(`${API_ENDPOINT}/info`).then((r) => r.json());

    return {
        props: {
            ssg: {
                count: approximate_member_count,
                info,
            },
        },
        revalidate: 10,
    };
}

export function Home() {
    const props = useSSG();
    return (
        <HomePage
            serverMembers={props?.count}
            usedBy={props.info.guilds}
        />
    );
}

<Home />
